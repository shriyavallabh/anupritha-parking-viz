Connect to Google Sheets (live sync setup)

Currently running in offline mode — new submissions are stored in your browser. Follow these steps to connect to Google Sheets so submissions sync live across all devices.

  1. Create a new Google Sheet at sheets.google.com.
  2. Rename the first tab to Vehicle Registry. Add a second tab: Form_Submissions.
  3. In the Vehicle Registry tab, paste the CSV data from the section below (existing Excel data).
  4. Open Extensions > Apps Script and paste this code:
function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);
  // Write to Form_Submissions
  ss.getSheetByName('Form_Submissions').appendRow([
    new Date(), data.flat, data.name,
    data.vehicleType, data.makeModel, data.regNumber
  ]);
  // Also add to Vehicle Registry
  ss.getSheetByName('Vehicle Registry').appendRow([
    data.flat, data.name, '', 'Owner',
    data.vehicleType, data.makeModel, data.regNumber, 'Active', new Date()
  ]);
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
